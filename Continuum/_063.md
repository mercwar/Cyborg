<section>
  <h2>Overview</h2>
  <p>
    The CBORD Build System orchestrates the full compiler chain:
  </p>
  <ul>
    <li>Parse CBORD / CYBORD sources</li>
    <li>Generate CBORD_IR</li>
    <li>Run optimizer passes</li>
    <li>Emit C / C++ / Java / ASM</li>
    <li>Compile with external toolchains</li>
    <li>Link objects into final binaries</li>
  </ul>
</section>

<section>
  <h2>CBORD_BUILD.cbs</h2>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_BUILD.cbs */

CBORD_BUILD_SYSTEM
{
    CONFIG {
        CC: "gcc"
        CXX: "g++"
        JAVA: "javac"
        ASM: "nasm"
        OUT_DIR: "build"
    }

    TARGETS: []

    ADD_TARGET(name, src, kind) {
        TARGETS.push({ name: name, src: src, kind: kind })
    }

    /* -------------------------------------- */
    /* Pipeline Steps                          */
    /* -------------------------------------- */

    STEP PARSE(src) {
        ir = CBORD_PARSER.parse(src)
        return ir
    }

    STEP OPTIMIZE(ir) {
        CBORD_OPTIMIZER.run(ir)
        return ir
    }

    STEP EMIT(ir, kind) {
        switch(kind) {
            case "C":    return CYBORD_EMITTER.emit_c(ir)
            case "CPP":  return CYBORD_EMITTER.emit_cpp(ir)
            case "JAVA": return CYBORD_EMITTER.emit_java(ir)
            case "ASM":  return CYBORD_EMITTER.emit_asm(ir)
        }
    }

    STEP COMPILE_C(path, out) {
        sh(CONFIG.CC + " -c " + path + " -o " + out)
    }

    STEP COMPILE_CPP(path, out) {
        sh(CONFIG.CXX + " -c " + path + " -o " + out)
    }

    STEP COMPILE_JAVA(path) {
        sh(CONFIG.JAVA + " " + path)
    }

    STEP ASSEMBLE(path, out) {
        sh(CONFIG.ASM + " -f win64 " + path + " -o " + out)
    }

    STEP LINK(objects, out) {
        CBORD_LINKER.OBJECTS = objects
        CBORD_LINKER.LINK()
        /* assumes linker writes 'a.out' or given name */
    }

    /* -------------------------------------- */
    /* Build All                               */
    /* -------------------------------------- */

    BUILD_ALL {
        objects = []

        for (t in TARGETS) {
            ir = PARSE(t.src)
            ir = OPTIMIZE(ir)
            code = EMIT(ir, t.kind)

            src_path = CONFIG.OUT_DIR + "/" + t.name + "." + ext_for_kind(t.kind)
            obj_path = CONFIG.OUT_DIR + "/" + t.name + ".o"

            write_file(src_path, code)

            switch(t.kind) {
                case "C":    COMPILE_C(src_path, obj_path);   break
                case "CPP":  COMPILE_CPP(src_path, obj_path); break
                case "ASM":  ASSEMBLE(src_path, obj_path);    break
                case "JAVA": COMPILE_JAVA(src_path);          break
            }

            if (t.kind != "JAVA") {
                objects.push(obj_path)
            }
        }

        if (objects.length &gt; 0) {
            LINK(objects, CONFIG.OUT_DIR + "/a.out")
        }
    }

    ext_for_kind(kind) {
        switch(kind) {
            case "C":    return "c"
            case "CPP":  return "cpp"
            case "JAVA": return "java"
            case "ASM":  return "asm"
        }
    }
}
</code></pre>
