// ============================================================
// CYBORG CORE (patched for STARFIRE)
// ============================================================

const CYBORG = (() => {
  const state = {
    merged_events: [],
    last_update: null
  };

  const now = () => new Date().toISOString();

  function fire(type, payload = {}) {
    const evt = { layer: "CYBORG", type, payload, timestamp: now() };
    state.merged_events.push(evt);
    state.last_update = now();
    return evt;
  }

  // STARFIRE bridge: canonical continuum event
  function toContinuum(type, payload = {}) {
    return {
      source_layer: "CYBORG",
      event_type: type,
      payload,
      timestamp: now()
    };
  }

  return {
    fire,
    toContinuum,
    getState: () => JSON.parse(JSON.stringify(state))
  };
})();
