# **CBORD_LOWERING_RULES.irf**  
*(CBORD → C → ASM lowering specification)*

```txt
/* AIFVS-ARTIFACT */
/* FILE: CBORD_LOWERING_RULES.irf */

CBORD_LOWERING_RULES
{
    /* Expression Lowering */
    LOWER BIN_OP {
        +  → C: (left + right), ASM: add
        -  → C: (left - right), ASM: sub
        *  → C: (left * right), ASM: imul
        /  → C: (left / right), ASM: idiv
        %  → C: (left % right), ASM: idiv + mod
        == → C: (left == right), ASM: cmp + je
        != → C: (left != right), ASM: cmp + jne
        <  → C: (left < right), ASM: cmp + jl
        >  → C: (left > right), ASM: cmp + jg
    }

    LOWER UN_OP {
        - → C: (-expr), ASM: neg
        ! → C: (!expr), ASM: cmp + je
        ~ → C: (~expr), ASM: not
    }

    /* Memory Lowering */
    LOWER LOAD {
        C: var
        ASM: mov reg, [addr]
    }

    LOWER STORE {
        C: var = value
        ASM: mov [addr], reg
    }

    /* Control Flow Lowering */
    LOWER IF {
        C: if(cond) { then } else { else }
        ASM:
            cmp cond
            je THEN_LABEL
            jmp ELSE_LABEL
    }

    LOWER WHILE {
        C: while(cond) { body }
        ASM:
            LOOP_START:
                cmp cond
                jne LOOP_END
                body
                jmp LOOP_START
            LOOP_END:
    }

    LOWER FOR {
        C: for(init; cond; step) { body }
        ASM:
            init
            LOOP_START:
                cmp cond
                jne LOOP_END
                body
                step
                jmp LOOP_START
            LOOP_END:
    }

    /* Function Lowering */
    LOWER FUNCTION {
        C: standard C function
        ASM:
            push rbp
            mov rbp, rsp
            ...body...
            pop rbp
            ret
    }

    /* Return Lowering */
    LOWER RETURN {
        C: return expr
        ASM: mov eax, expr ; ret
    }

    /* Inline ASM */
    LOWER ASM_INLINE {
        C: __asm__(text)
        ASM: text
    }

    /* Direct C Emission */
    LOWER EMIT_C {
        C: c_text
        ASM: (none)
    }
}
```
