<section>
  <h2>1. CBORD_PARSER.cbp</h2>
  <p>The CBORD parser reads CBORD source and produces CBORD_IR nodes.</p>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_PARSER.cbp */

PARSER {
    TOKENS {
        KEYWORDS: [CBORD, BEGIN, END, IF, WHILE, FOR, RETURN, ASM]
        TYPES: [int, float, char, void]
        SYMBOLS: [+, -, *, /, =, ==, !=, &lt;, &gt;, {, }, (, ), ;]
    }

    RULE program {
        CBORD BEGIN
        statements*
        CBORD END
    }

    RULE statement {
        decl | assign | call | if | while | for | return | asm_inline
    }

    RULE decl {
        type IDENT "=" expr ";"
    }

    RULE assign {
        IDENT "=" expr ";"
    }

    RULE call {
        IDENT "(" args ")" ";"
    }

    RULE if {
        IF "(" expr ")" block (ELSE block)?
    }

    RULE while {
        WHILE "(" expr ")" block
    }

    RULE for {
        FOR "(" init ";" cond ";" step ")" block
    }

    RULE return {
        RETURN expr ";"
    }

    RULE asm_inline {
        ASM "{" TEXT "}"
    }
}
</code></pre>
</section>

<!-- FILE 3: LOWERING RULES -->
<section>
  <h2>3. CBORD_LOWERING_RULES.irf</h2>
  <p>Rules for lowering CBORD_IR into C or ASM.</p>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_LOWERING_RULES.irf */

CBORD_LOWERING_RULES
{
    LOWER BIN_OP {
        + → C: (left + right), ASM: add
        - → C: (left - right), ASM: sub
        * → C: (left * right), ASM: imul
        / → C: (left / right), ASM: idiv
        == → C: (left == right), ASM: cmp + je
        != → C: (left != right), ASM: cmp + jne
    }

    LOWER IF {
        C: if(cond) { then } else { else }
        ASM:
            cmp cond
            je THEN_LABEL
            jmp ELSE_LABEL
    }

    LOWER WHILE {
        C: while(cond) { body }
        ASM:
            LOOP_START:
                cmp cond
                jne LOOP_END
                body
                jmp LOOP_START
            LOOP_END:
    }

    LOWER FUNCTION {
        C: standard C function
        ASM:
            push rbp
            mov rbp, rsp
            ...body...
            pop rbp
            ret
    }

    LOWER ASM_INLINE {
        C: __asm__(text)
        ASM: text
    }

    LOWER EMIT_C {
        C: c_text
        ASM: (none)
    }
}
</code></pre>
</section>

<!-- FILE 4: ASM EMITTER -->
<section>
  <h2>4. CBORD_ASM_EMITTER.abf</h2>
  <p>The ASM emitter converts lowered IR into real x86‑64 assembly.</p>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_ASM_EMITTER.abf */

CBORD_ASM_EMITTER
{
    TARGET: x86_64_WIN64

    EMIT FUNCTION_BEGIN(name) {
        print("global " + name)
        print(name + ":")
        print("    push rbp")
        print("    mov rbp, rsp")
    }

    EMIT FUNCTION_END {
        print("    pop rbp")
        print("    ret")
    }

    EMIT BIN_OP(op, left, right) {
        switch(op) {
            case "+": print("mov rax, " + left)
                      print("add rax, " + right)
                      break

            case "-": print("mov rax, " + left)
                      print("sub rax, " + right)
                      break

            case "*": print("mov rax, " + left)
                      print("imul rax, " + right)
                      break

            case "/": print("mov rax, " + left)
                      print("cqo")
                      print("idiv " + right)
                      break
        }
    }

    EMIT LOAD(addr) {
        print("mov rax, [" + addr + "]")
    }

    EMIT STORE(addr, value) {
        print("mov [" + addr + "], " + value)
    }

    EMIT RETURN(expr) {
        print("mov rax, " + expr)
        print("pop rbp")
        print("ret")
    }
}
</code></pre>
