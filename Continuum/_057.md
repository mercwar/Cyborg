  <p>
    The CYBORD emitter consumes <strong>CBORD_IR</strong> and produces:
  </p>
  <ul>
    <li>C source</li>
    <li>C++ source</li>
    <li>Java source</li>
    <li>ASM (inline or full)</li>
  </ul>

    CYBORD is the “polyglot tier” of the Continuum — the layer where one IR becomes many languages.

  <h2>CYBORD_EMITTER.cyb</h2>
  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CYBORD_EMITTER.cyb */

CYBORD_EMITTER
{
    TARGETS: [C, CPP, JAVA, ASM]

    /* Shared helpers */
    EMIT IDENT(name) {
        return sanitize(name)
    }

    EMIT TYPE(t) {
        switch(t) {
            case INT: return "int"
            case FLOAT: return "float"
            case CHAR: return "char"
            case VOID: return "void"
        }
    }

    /* ------------------------------ */
    /* C EMISSION                     */
    /* ------------------------------ */
    EMIT_C FUNCTION(fn) {
        print(TYPE(fn.ret_type) + " " + fn.name + "(")
        emit_params(fn.params)
        print(") {")
        emit_block(fn.body)
        print("}")
    }

    EMIT_C BIN_OP(op, left, right) {
        print("(" + left + " " + op + " " + right + ")")
    }

    EMIT_C RETURN(expr) {
        print("return " + expr + ";")
    }

    /* ------------------------------ */
    /* C++ EMISSION                   */
    /* ------------------------------ */
    EMIT_CPP FUNCTION(fn) {
        print(TYPE(fn.ret_type) + " " + fn.name + "(")
        emit_params(fn.params)
        print(") {")
        emit_block(fn.body)
        print("}")
    }

    EMIT_CPP MAIN_WRAPPER {
        print("#include &lt;iostream&gt;")
        print("int main() {")
        print("    std::cout &lt;&lt; add(1,2) &lt;&lt; std::endl;")
        print("    return 0;")
        print("}")
    }

    /* ------------------------------ */
    /* JAVA EMISSION                  */
    /* ------------------------------ */
    EMIT_JAVA FUNCTION(fn) {
        print("static " + TYPE(fn.ret_type) + " " + fn.name + "(")
        emit_params_java(fn.params)
        print(") {")
        emit_block_java(fn.body)
        print("}")
    }

    EMIT_JAVA MAIN_WRAPPER {
        print("public class Main {")
        print("    public static void main(String[] args) {")
        print("        System.out.println(add(1,2));")
        print("    }")
        print("}")
    }

    /* ------------------------------ */
    /* ASM EMISSION                   */
    /* ------------------------------ */
    EMIT_ASM FUNCTION(fn) {
        print("global " + fn.name)
        print(fn.name + ":")
        print("    push rbp")
        print("    mov rbp, rsp")
        emit_block_asm(fn.body)
        print("    pop rbp")
        print("    ret")
    }

    EMIT_ASM BIN_OP(op, left, right) {
        switch(op) {
            case "+": print("    mov rax, " + left)
                      print("    add rax, " + right)
                      break

            case "-": print("    mov rax, " + left)
                      print("    sub rax, " + right)
                      break

            case "*": print("    mov rax, " + left)
                      print("    imul rax, " + right)
                      break

            case "/": print("    mov rax, " + left)
                      print("    cqo")
                      print("    idiv " + right)
                      break
        }
    }

    EMIT_ASM RETURN(expr) {
        print("    mov rax, " + expr)
        print("    pop rbp")
        print("    ret")
    }
}
</code></pre>
