// ============================================================
// STARFIRE CONTINUUM BRIDGE
// ============================================================

const STARFIRE_BRIDGE = (() => {
  const now = () => new Date().toISOString();

  function fromLayer(layer, type, payload) {
    let continuum;
    switch (layer) {
      case "CBORD":
        continuum = CBORD.toContinuum(type, payload);
        break;
      case "CYBORD":
        continuum = CYBORD.toContinuum(type, payload);
        break;
      case "CBORG":
        continuum = CBORG.toContinuum(type, payload);
        break;
      case "CYBORG":
        continuum = CYBORG.toContinuum(type, payload);
        break;
      default:
        continuum = { source_layer: layer, event_type: type, payload, timestamp: now() };
    }

    // Ingest into STARFIRE as a continuum ascension
    return STARFIRE.ingestContinuumEvent(continuum);
  }

  return {
    fromLayer
  };
})();
