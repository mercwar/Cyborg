<section>
  <h2>CBORD_OPTIMIZER.cof</h2>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_OPTIMIZER.cof */

CBORD_OPTIMIZER
{
    PASS CONSTANT_FOLD {
        RULE BIN_OP(op, CONST(a), CONST(b)) {
            eval = fold(op, a, b)
            REPLACE WITH CONST(eval)
        }
    }

    PASS DEAD_CODE_ELIM {
        RULE ASSIGN(VAR(x), expr) {
            if (!is_used_later(x)) {
                REMOVE NODE
            }
        }
    }

    PASS COPY_PROP {
        RULE ASSIGN(VAR(x), VAR(y)) {
            propagate(x, y)
        }
    }

    PASS PHI_CLEANUP {
        RULE PHI(target, incoming[]) {
            if (incoming.size == 1) {
                REPLACE WITH ASSIGN(target, incoming[0].value)
            }
        }
    }

    FUNCTION fold(op, a, b) {
        switch(op) {
            case "+": return a + b
            case "-": return a - b
            case "*": return a * b
            case "/": return b != 0 ? a / b : a /* keep as-is if div by 0 */
            case "==": return a == b
            case "!=": return a != b
        }
    }
}
</code></pre>
</section>
