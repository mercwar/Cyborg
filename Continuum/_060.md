<section>
  <h2>Overview</h2>
  <p>
    The CBORD Linker merges:
  </p>
  <ul>
    <li>ASM object files</li>
    <li>C object files</li>
    <li>Runtime objects</li>
    <li>External libraries</li>
  </ul>
  <p>
    It resolves symbols, patches relocations, and produces a final executable or library.
  </p>
</section>

<section>
  <h2>CBORD_LINKER.clf</h2>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_LINKER.clf */

CBORD_LINKER
{
    INPUTS {
        OBJECTS: []
        RUNTIME: "CBORD_RUNTIME.o"
        LIBS: []
    }

    ADD_OBJECT(path) {
        OBJECTS.push(path)
    }

    ADD_LIBRARY(path) {
        LIBS.push(path)
    }

    /* -------------------------------------- */
    /* Symbol Table                            */
    /* -------------------------------------- */

    SYMBOL_TABLE {
        GLOBALS: {}
        EXTERNS: {}
    }

    RESOLVE_SYMBOL(name, source) {
        if (SYMBOL_TABLE.GLOBALS[name]) return SYMBOL_TABLE.GLOBALS[name]
        if (SYMBOL_TABLE.EXTERNS[name]) return SYMBOL_TABLE.EXTERNS[name]
        ERROR("Unresolved symbol: " + name)
    }

    /* -------------------------------------- */
    /* Relocation Processing                   */
    /* -------------------------------------- */

    PROCESS_RELOC(obj, reloc) {
        target = RESOLVE_SYMBOL(reloc.symbol, obj)
        patch(obj.binary, reloc.offset, target.address)
    }

    /* -------------------------------------- */
    /* Linking Pipeline                        */
    /* -------------------------------------- */

    LINK {
        /* Load all objects */
        for (obj in OBJECTS) {
            load(obj)
            merge_symbols(obj)
            process_relocations(obj)
        }

        /* Load runtime */
        load(RUNTIME)
        merge_symbols(RUNTIME)
        process_relocations(RUNTIME)

        /* Load external libs */
        for (lib in LIBS) {
            load(lib)
            merge_symbols(lib)
            process_relocations(lib)
        }

        /* Merge all binaries */
        final = merge_all_binaries(OBJECTS, RUNTIME, LIBS)

        /* Emit executable */
        write_executable("a.out", final)
    }

    /* -------------------------------------- */
    /* Error Handling                          */
    /* -------------------------------------- */

    ERROR(msg) {
        print("CBORD LINK ERROR: " + msg)
        abort()
    }
}
</code></pre>
</section>
