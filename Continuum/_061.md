<section>
  <h2>Overview</h2>
  <p>
    The CYBORD Standard Library provides a unified set of helpers that can be emitted into:
  </p>
  <ul>
    <li>C (header + implementation)</li>
    <li>C++ (namespace wrapper)</li>
    <li>Java (static utility class)</li>
    <li>ASM (callable helpers)</li>
  </ul>
</section>

<section>
  <h2>CYBORD_STD.cys</h2>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CYBORD_STD.cys */

CYBORD_STD
{
    /* -------------------------------------- */
    /* C EMISSION                             */
    /* -------------------------------------- */

    EMIT_C HEADER {
        print("#ifndef CYBORD_STD_H")
        print("#define CYBORD_STD_H")
        print("#include &lt;stdio.h&gt;")
        print("#include &lt;stdlib.h&gt;")
        print("int cy_add(int a, int b);")
        print("int cy_mul(int a, int b);")
        print("void cy_print_int(int x);")
        print("#endif")
    }

    EMIT_C IMPLEMENTATION {
        print("#include \"CYBORD_STD.h\"")
        print("int cy_add(int a, int b) { return a + b; }")
        print("int cy_mul(int a, int b) { return a * b; }")
        print("void cy_print_int(int x) { printf(\"%d\\n\", x); }")
    }

    /* -------------------------------------- */
    /* C++ EMISSION                           */
    /* -------------------------------------- */

    EMIT_CPP HEADER {
        print("#ifndef CYBORD_STD_HPP")
        print("#define CYBORD_STD_HPP")
        print("#include &lt;iostream&gt;")
        print("namespace cy {")
        print("    inline int add(int a, int b) { return a + b; }")
        print("    inline int mul(int a, int b) { return a * b; }")
        print("    inline void print_int(int x) { std::cout &lt;&lt; x &lt;&lt; std::endl; }")
        print("}")
        print("#endif")
    }

    /* -------------------------------------- */
    /* JAVA EMISSION                          */
    /* -------------------------------------- */

    EMIT_JAVA CLASS {
        print("public final class CyStd {")
        print("    private CyStd() {}")
        print("    public static int add(int a, int b) { return a + b; }")
        print("    public static int mul(int a, int b) { return a * b; }")
        print("    public static void printInt(int x) { System.out.println(x); }")
        print("}")
    }

    /* -------------------------------------- */
    /* ASM EMISSION                           */
    /* -------------------------------------- */

    EMIT_ASM HELPERS {
        print("global cy_add_asm")
        print("cy_add_asm:")
        print("    push rbp")
        print("    mov rbp, rsp")
        print("    mov eax, edi")
        print("    add eax, esi")
        print("    pop rbp")
        print("    ret")

        print("global cy_mul_asm")
        print("cy_mul_asm:")
        print("    push rbp")
        print("    mov rbp, rsp")
        print("    mov eax, edi")
        print("    imul eax, esi")
        print("    pop rbp")
        print("    ret")
    }
}
</code></pre>
</section>

<section>
  <h2>Usage Examples</h2>

  <h3>C</h3>
  <pre><code>#include "CYBORD_STD.h"

int main(void) {
    int x = cy_add(10, 20);
    cy_print_int(x);
    return 0;
}</code></pre>

  <h3>C++</h3>
  <pre><code>#include "CYBORD_STD.hpp"
using namespace cy;

int main() {
    int x = add(10, 20);
    print_int(x);
    return 0;
}</code></pre>

  <h3>Java</h3>
  <pre><code>public class Main {
    public static void main(String[] args) {
        int x = CyStd.add(10, 20);
        CyStd.printInt(x);
    }
}</code></pre>

  <h3>ASM</h3>
  <pre><code>extern cy_add_asm

global main
main:
    push rbp
    mov rbp, rsp

    mov edi, 10
    mov esi, 20
    call cy_add_asm

    pop rbp
    ret
</code></pre>
