// ============================================================
// CBORG CORE (patched for STARFIRE)
// ============================================================

const CBORG = (() => {
  const state = {
    dispatch_events: [],
    last_update: null
  };

  const now = () => new Date().toISOString();

  function dispatch(type, payload = {}) {
    const evt = { layer: "CBORG", type, payload, timestamp: now() };
    state.dispatch_events.push(evt);
    state.last_update = now();
    return evt;
  }

  // STARFIRE bridge
  function toContinuum(type, payload = {}) {
    return {
      source_layer: "CBORG",
      event_type: type,
      payload,
      timestamp: now()
    };
  }

  return {
    dispatch,
    toContinuum,
    getState: () => JSON.parse(JSON.stringify(state))
  };
})();
