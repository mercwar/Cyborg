  <h2>CBORD_RUNTIME.crt</h2>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_RUNTIME.crt */

#ifndef CBORD_RUNTIME_H
#define CBORD_RUNTIME_H

/* -------------------------------------- */
/* Memory Helpers                         */
/* -------------------------------------- */

#include &lt;stdlib.h&gt;
#include &lt;string.h&gt;

static inline void* cbord_alloc(size_t n)
{
    void* p = malloc(n);
    if (!p) abort();
    return p;
}

static inline void cbord_free(void* p)
{
    free(p);
}

static inline void* cbord_zero(void* p, size_t n)
{
    return memset(p, 0, n);
}

/* -------------------------------------- */
/* String Helpers                         */
/* -------------------------------------- */

static inline size_t cbord_strlen(const char* s)
{
    size_t n = 0;
    while (s[n]) n++;
    return n;
}

static inline char* cbord_strcpy(char* dst, const char* src)
{
    char* d = dst;
    while ((*d++ = *src++));
    return dst;
}

static inline char* cbord_strdup(const char* src)
{
    size_t n = cbord_strlen(src) + 1;
    char* d = cbord_alloc(n);
    cbord_strcpy(d, src);
    return d;
}

/* -------------------------------------- */
/* Arithmetic Helpers                      */
/* -------------------------------------- */

static inline int cbord_add(int a, int b)
{
    return a + b;
}

static inline int cbord_sub(int a, int b)
{
    return a - b;
}

static inline int cbord_mul(int a, int b)
{
    return a * b;
}

static inline int cbord_div(int a, int b)
{
    if (b == 0) abort();
    return a / b;
}

/* -------------------------------------- */
/* ASM-Friendly Entry Points               */
/* -------------------------------------- */

#if defined(__x86_64__)

static inline int cbord_add_asm(int a, int b)
{
    int r;
    __asm__(
        "mov %1, %%eax\n"
        "add %2, %%eax\n"
        "mov %%eax, %0\n"
        : "=r"(r)
        : "r"(a), "r"(b)
        : "%eax"
    );
    return r;
}

#endif /* __x86_64__ */

/* -------------------------------------- */
/* OS Abstraction                          */
/* -------------------------------------- */

static inline void cbord_exit(int code)
{
    exit(code);
}

#endif /* CBORD_RUNTIME_H */
