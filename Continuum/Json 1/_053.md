```txt
/* AIFVS-ARTIFACT */
/* FILE: CBORD_ASM_EMITTER.abf */

CBORD_ASM_EMITTER
{
    TARGET: x86_64_WIN64
    CALLING_CONVENTION: Win64

    /* Register Allocation */
    REGISTERS {
        INT: [rax, rbx, rcx, rdx, rsi, rdi]
        TEMP: [r8, r9, r10, r11]
        STACK: rbp, rsp
    }

    /* Function Prologue / Epilogue */
    EMIT FUNCTION_BEGIN(name) {
        print("global " + name)
        print(name + ":")
        print("    push rbp")
        print("    mov rbp, rsp")
    }

    EMIT FUNCTION_END {
        print("    pop rbp")
        print("    ret")
    }

    /* Expressions */
    EMIT BIN_OP(op, left, right) {
        switch(op) {
            case "+": print("    mov rax, " + left)
                      print("    add rax, " + right)
                      break

            case "-": print("    mov rax, " + left)
                      print("    sub rax, " + right)
                      break

            case "*": print("    mov rax, " + left)
                      print("    imul rax, " + right)
                      break

            case "/": print("    mov rax, " + left)
                      print("    cqo")
                      print("    idiv " + right)
                      break

            case "==": print("    cmp " + left + ", " + right)
                       print("    sete al")
                       print("    movzx rax, al")
                       break
        }
    }

    /* Memory */
    EMIT LOAD(addr) {
        print("    mov rax, [" + addr + "]")
    }

    EMIT STORE(addr, value) {
        print("    mov [" + addr + "], " + value)
    }

    /* Control Flow */
    EMIT IF(cond, then_lbl, else_lbl) {
        print("    cmp " + cond.left + ", " + cond.right)
        print("    je " + then_lbl)
        print("    jmp " + else_lbl)
    }

    EMIT WHILE(start_lbl, end_lbl, cond) {
        print(start_lbl + ":")
        print("    cmp " + cond.left + ", " + cond.right)
        print("    jne " + end_lbl)
    }

    /* Inline ASM */
    EMIT ASM_INLINE(text) {
        print(text)
    }

    /* Return */
    EMIT RETURN(expr) {
        print("    mov rax, " + expr)
        print("    pop rbp")
        print("    ret")
    }
}
```
