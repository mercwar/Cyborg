// CBORD CORE — STARFIRE-PATCHED
// ============================================================

const CBORD = (() => {
  const state = {
    runtime_events: [],
    last_update: null
  };

  const now = () => new Date().toISOString();

  function emit(type, payload = {}) {
    const evt = { layer: "CBORD", type, payload, timestamp: now() };
    state.runtime_events.push(evt);
    state.last_update = now();
    return evt;
  }

  function toContinuum(type, payload = {}) {
    return {
      source_layer: "CBORD",
      event_type: type,
      payload,
      timestamp: now()
    };
  }

  return {
    emit,
    toContinuum,
    getState: () => JSON.parse(JSON.stringify(state))
  };
})();
