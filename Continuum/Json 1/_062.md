  <h2>Overview</h2>
  <p>
    The CBORD Debugger provides:
  </p>
  <ul>
    <li>IR tree inspection</li>
    <li>ASM instruction tracing</li>
    <li>Breakpoints on IR nodes</li>
    <li>Runtime variable watch</li>
    <li>Step‑through execution</li>
  </ul>
  <p>
    It is designed to work with both CBORD_IR and ASM emitted by the CBORD ASM Emitter.
  </p>
</section>

<section>
  <h2>CBORD_DEBUGGER.cdb</h2>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_DEBUGGER.cdb */

CBORD_DEBUGGER
{
    /* -------------------------------------- */
    /* IR Inspection                           */
    /* -------------------------------------- */

    INSPECT_IR(node) {
        print("IR NODE: " + node.type)
        for (field in node.fields) {
            print("  " + field.name + ": " + field.value)
        }
    }

    DUMP_IR(module) {
        for (fn in module.functions) {
            print("FUNCTION: " + fn.name)
            for (block in fn.blocks) {
                print("  BLOCK: " + block.label)
                for (stmt in block.statements) {
                    INSPECT_IR(stmt)
                }
            }
        }
    }

    /* -------------------------------------- */
    /* ASM Tracing                             */
    /* -------------------------------------- */

    TRACE_ASM(line) {
        print("ASM: " + line.text)
    }

    DUMP_ASM(program) {
        for (line in program.asm_lines) {
            TRACE_ASM(line)
        }
    }

    /* -------------------------------------- */
    /* Breakpoints                             */
    /* -------------------------------------- */

    BREAKPOINTS: []

    SET_BREAKPOINT_IR(node_type) {
        BREAKPOINTS.push({ type: "IR", match: node_type })
    }

    SET_BREAKPOINT_ASM(mnemonic) {
        BREAKPOINTS.push({ type: "ASM", match: mnemonic })
    }

    CHECK_BREAKPOINT(event) {
        for (bp in BREAKPOINTS) {
            if (bp.type == "IR" && event.node.type == bp.match) {
                print("BREAKPOINT HIT (IR): " + bp.match)
                debugger_pause()
            }
            if (bp.type == "ASM" && event.asm.mnemonic == bp.match) {
                print("BREAKPOINT HIT (ASM): " + bp.match)
                debugger_pause()
            }
        }
    }

    /* -------------------------------------- */
    /* Runtime Watch                            */
    /* -------------------------------------- */

    WATCHLIST: {}

    WATCH(var_name) {
        WATCHLIST[var_name] = true
    }

    UPDATE_WATCH(var_name, value) {
        if (WATCHLIST[var_name]) {
            print("WATCH: " + var_name + " = " + value)
        }
    }

    /* -------------------------------------- */
    /* Execution Control                        */
    /* -------------------------------------- */

    STEP_IR(node) {
        CHECK_BREAKPOINT({ type: "IR", node: node })
        INSPECT_IR(node)
    }

    STEP_ASM(line) {
        CHECK_BREAKPOINT({ type: "ASM", asm: line })
        TRACE_ASM(line)
    }

    debugger_pause() {
        print("DEBUGGER PAUSED — press any key to continue")
        wait_for_key()
    }
}
</code></pre>
