// ============================================================
// STARFIRE ASTRAL UI LAYER
// ============================================================
// Demon — this is the visual command surface for STARFIRE.
// It binds:
//   - Control Deck
//   - Starforge Engine
//   - Diagnostics
//   - Memory Core
//   - Celestial Archive
//   - Forge Runtime
//
// This UI layer renders astral panels, buttons, logs, and
// constellation visualizers.
// ============================================================

const STARFIRE_UI = (() => {

  // ------------------------------------------------------------
  // DOM HELPERS
  // ------------------------------------------------------------
  const $ = sel => document.querySelector(sel);
  const $$ = sel => Array.from(document.querySelectorAll(sel));

  // ------------------------------------------------------------
  // PANELS
  // ------------------------------------------------------------
  let navPanel, eventPanel, diagPanel, forgePanel, memoryPanel, archivePanel, outputPanel;

  function bindPanels() {
    navPanel     = $("#starfire-nav-panel");
    eventPanel   = $("#starfire-event-panel");
    diagPanel    = $("#starfire-diagnostics-panel");
    forgePanel   = $("#starfire-forge-panel");
    memoryPanel  = $("#starfire-memory-panel");
    archivePanel = $("#starfire-archive-panel");
    outputPanel  = $("#starfire-output");
  }

  // ------------------------------------------------------------
  // RENDER JSON
  // ------------------------------------------------------------
  function render(obj) {
    outputPanel.textContent = JSON.stringify(obj, null, 2);
  }

  // ------------------------------------------------------------
  // ACTION HANDLERS
  // ------------------------------------------------------------
  function handleTraverseRoute() {
    const from = $("#nav-from").value;
    const to   = $("#nav-to").value;

    const result = STARFIRE.orchestrate("astral_traverse", { from, to });
    render(result);
  }

  function handleForgeStar() {
    const name = $("#forge-star-name").value;
    const type = $("#forge-star-type").value;

    const result = STARFIRE_FORGE_RUNTIME.RUNTIME_OPERATIONS.FORGE_STAR({
      name,
      type,
      meta: { forged_by: "UI" }
    });

    render(result);
  }

  function handleForgeEvent() {
    const origin = $("#event-origin").value;
    const dest   = $("#event-dest").value;

    const result = STARFIRE_FORGE_RUNTIME.RUNTIME_OPERATIONS.FORGE_EVENT({
      origin_star: origin,
      destination_star: dest,
      type: "ASTRAL_SIGNAL",
      mode: "DIRECT",
      payload: { ui: true }
    });

    render(result);
  }

  function handleDiagnostics() {
    const star = $("#diag-star").value;

    const result = STARFIRE_ASTRAL_DIAGNOSTICS.DIAGNOSTIC_OPERATIONS.SCAN_STAR({
      star
    });

    render(result);
  }

  function handleMemoryRead() {
    const domain = $("#memory-domain").value;

    const result = STARFIRE_CELESTIAL_MEMORY_CORE.MEMORY_OPERATIONS.READ_DOMAIN({
      domain
    });

    render(result);
  }

  function handleArchiveSnapshot() {
    const result = STARFIRE_CELESTIAL_ARCHIVE.ARCHIVE_OPERATIONS.CREATE_SNAPSHOT({
      state: STARFIRE.getState(),
      meta: { captured_by: "UI" }
    });

    render(result);
  }

  // ------------------------------------------------------------
  // WIRE BUTTONS
  // ------------------------------------------------------------
  function wireButtons() {
    $("#btn-traverse-route").addEventListener("click", handleTraverseRoute);
    $("#btn-forge-star").addEventListener("click", handleForgeStar);
    $("#btn-forge-event").addEventListener("click", handleForgeEvent);
    $("#btn-run-diagnostics").addEventListener("click", handleDiagnostics);
    $("#btn-memory-read").addEventListener("click", handleMemoryRead);
    $("#btn-archive-snapshot").addEventListener("click", handleArchiveSnapshot);
  }

  // ------------------------------------------------------------
  // INIT
  // ------------------------------------------------------------
  function init() {
    bindPanels();
    wireButtons();
  }

  document.addEventListener("DOMContentLoaded", init);

  return {
    init,
    render
  };
})();
