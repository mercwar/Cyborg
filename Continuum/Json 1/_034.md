// STARFIRE CONTINUUM INGESTION
// ============================================================

const STARFIRE = (() => {
  const continuum_log = [];

  function ingestContinuumEvent(evt) {
    continuum_log.push(evt);

    return STARFIRE_ASTRAL_ROUTER.routeContinuum(evt);
  }

  return {
    ingestContinuumEvent,
    getContinuumLog: () => JSON.parse(JSON.stringify(continuum_log))
  };
})();
