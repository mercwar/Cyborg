<section>
  <h2>Overview</h2>
  <p>
    The CBORD Virtual Machine (CBVM) provides:
  </p>
  <ul>
    <li>Execution of CBORD_IR instructions</li>
    <li>Register + stack simulation</li>
    <li>Memory model (heap + globals)</li>
    <li>Inline ASM execution hooks</li>
    <li>Breakpoints + tracing</li>
    <li>Deterministic replay</li>
  </ul>
  <p>
    It is the “software CPU” for CBORD programs.
  </p>
</section>

  <h2>CBORD_VM.cvm</h2>

  <pre><code>/* AIFVS-ARTIFACT */
/* FILE: CBORD_VM.cvm */

CBORD_VM
{
    /* -------------------------------------- */
    /* Registers                               */
    /* -------------------------------------- */

    REGISTERS {
        r0: 0
        r1: 0
        r2: 0
        r3: 0
        sp: 0
        bp: 0
        ip: 0
    }

    /* -------------------------------------- */
    /* Memory Model                            */
    /* -------------------------------------- */

    HEAP: []
    GLOBALS: {}

    ALLOC(size) {
        addr = HEAP.length
        HEAP.resize(addr + size)
        return addr
    }

    LOAD(addr) {
        return HEAP[addr]
    }

    STORE(addr, value) {
        HEAP[addr] = value
    }

    /* -------------------------------------- */
    /* Instruction Set                         */
    /* -------------------------------------- */

    INSTRUCTIONS {
        MOV:   (dst, src) => REGISTERS[dst] = src
        ADD:   (dst, src) => REGISTERS[dst] += src
        SUB:   (dst, src) => REGISTERS[dst] -= src
        MUL:   (dst, src) => REGISTERS[dst] *= src
        DIV:   (dst, src) => REGISTERS[dst] /= src

        LOAD:  (dst, addr) => REGISTERS[dst] = LOAD(addr)
        STORE: (addr, src) => STORE(addr, REGISTERS[src])

        JMP:   (target) => REGISTERS.ip = target
        JZ:    (target) => if (REGISTERS.r0 == 0) REGISTERS.ip = target
        JNZ:   (target) => if (REGISTERS.r0 != 0) REGISTERS.ip = target

        CALL:  (target) => {
            push(REGISTERS.ip)
            REGISTERS.ip = target
        }

        RET: () => {
            REGISTERS.ip = pop()
        }
    }

    /* -------------------------------------- */
    /* Stack Operations                         */
    /* -------------------------------------- */

    STACK: []

    push(value) {
        STACK.push(value)
    }

    pop() {
        return STACK.pop()
    }

    /* -------------------------------------- */
    /* Execution Loop                           */
    /* -------------------------------------- */

    EXEC(program) {
        REGISTERS.ip = 0

        while (REGISTERS.ip < program.length) {
            instr = program[REGISTERS.ip]
            EXEC_INSTR(instr)
            REGISTERS.ip++
        }
    }

    EXEC_INSTR(instr) {
        opcode = instr.op
        args = instr.args
        INSTRUCTIONS[opcode](...args)
    }

    /* -------------------------------------- */
    /* Inline ASM Hook                          */
    /* -------------------------------------- */

    EXEC_ASM(text) {
        /* Pass-through to host ASM or simulate */
        print("ASM EXEC: " + text)
    }

    /* -------------------------------------- */
    /* Debugging                                */
    /* -------------------------------------- */

    TRACE(instr) {
        print("TRACE: " + instr.op + " " + instr.args)
    }

    BREAKPOINTS: []

    SET_BREAK(opcode) {
        BREAKPOINTS.push(opcode)
    }

    CHECK_BREAK(instr) {
        if (BREAKPOINTS.contains(instr.op)) {
            print("BREAKPOINT HIT: " + instr.op)
            debugger_pause()
        }
    }

    debugger_pause() {
        print("VM PAUSED — press key to continue")
        wait_for_key()
    }
}
</code></pre>
