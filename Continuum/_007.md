// ============================================================
// STARFIRE CORE
// ============================================================
// Demon — this is the first file of the STARFIRE layer.
// It sits ABOVE:
//   CBORD → CYBORD → CBORG → INCINTUUM → CYBORG
//
// STARFIRE is the astral-grade orchestrator that manages:
//   - Constellation nodes
//   - Star routes
//   - Astral events
//   - Continuum bindings
//
// ============================================================

const STARFIRE = (() => {

  // ------------------------------------------------------------
  // STARFIRE STATE
  // ------------------------------------------------------------
  const state = {
    constellation: {},
    routes: [],
    astral_events: [],
    continuum_bindings: [],
    lifecycle: {
      boot_time: null,
      last_update: null,
      status: "initializing"
    }
  };

  const now = () => new Date().toISOString();

  // ------------------------------------------------------------
  // DEFINE A STAR NODE
  // ------------------------------------------------------------
  function defineStar(name, meta = {}) {
    const star = {
      name,
      meta,
      created: now(),
      links: []
    };
    state.constellation[name] = star;
    state.lifecycle.last_update = now();
    return star;
  }

  // ------------------------------------------------------------
  // LINK STARS (astral route)
  // ------------------------------------------------------------
  function linkStars(from, to, type = "astral_route") {
    if (!state.constellation[from] || !state.constellation[to]) {
      throw new Error("Star not found");
    }

    const route = {
      from,
      to,
      type,
      timestamp: now()
    };

    state.routes.push(route);
    state.constellation[from].links.push(route);
    state.lifecycle.last_update = now();

    return route;
  }

  // ------------------------------------------------------------
  // INGEST A CONTINUUM EVENT (from CYBORG)
  // ------------------------------------------------------------
  function ingestContinuumEvent(event) {
    const astralEvent = {
      source: "continuum",
      payload: event,
      timestamp: now()
    };

    state.astral_events.push(astralEvent);
    state.lifecycle.last_update = now();

    return astralEvent;
  }

  // ------------------------------------------------------------
  // BIND CONTINUUM → STARFIRE
  // ------------------------------------------------------------
  function bindContinuumToStar(starName, bindingMeta = {}) {
    if (!state.constellation[starName]) {
      throw new Error("Star not found");
    }

    const binding = {
      star: starName,
      meta: bindingMeta,
      timestamp: now()
    };

    state.continuum_bindings.push(binding);
    state.lifecycle.last_update = now();

    return binding;
  }

  // ------------------------------------------------------------
  // STARFIRE ORCHESTRATE
  // ------------------------------------------------------------
  function orchestrate(eventType, payload) {
    // 1. Pull event from CYBORG
    const continuumEvent = CYBORG.fire(eventType, payload);

    // 2. Convert to astral event
    const astralEvent = ingestContinuumEvent(continuumEvent);

    // 3. Return STARFIRE-level event
    return {
      continuum: continuumEvent,
      astral: astralEvent,
      starfire_state: JSON.parse(JSON.stringify(state))
    };
  }

  // ------------------------------------------------------------
  // BOOT
  // ------------------------------------------------------------
  function boot() {
    state.lifecycle.boot_time = now();
    state.lifecycle.status = "starfire-running";
  }

  boot();

  // ------------------------------------------------------------
  // PUBLIC API
  // ------------------------------------------------------------
  return {
    defineStar,
    linkStars,
    ingestContinuumEvent,
    bindContinuumToStar,
    orchestrate,
    getState: () => JSON.parse(JSON.stringify(state))
  };
})();
