// ============================================================
// CYBORG DHTML UI LAYER
// ============================================================
// Demon — this file binds the CYBORG runtime to the HTML UI.
// It provides:
//   - Live event buttons
//   - Real-time logs
//   - Continuum state viewer
//   - CBORG orchestrator controls
//   - JSON inspector panels
// ============================================================

const CYBORG_UI = (() => {

  // ------------------------------------------------------------
  // DOM LOOKUP HELPERS
  // ------------------------------------------------------------
  const $ = sel => document.querySelector(sel);
  const $$ = sel => Array.from(document.querySelectorAll(sel));

  // ------------------------------------------------------------
  // ELEMENTS
  // ------------------------------------------------------------
  let logEl, stateEl, dispatchEl, gatewayEl;

  function bindElements() {
    logEl     = $("#cyborg-log");
    stateEl   = $("#cyborg-state");
    dispatchEl= $("#cyborg-dispatch");
    gatewayEl = $("#cyborg-gateway");
  }

  // ------------------------------------------------------------
  // RENDER HELPERS
  // ------------------------------------------------------------
  function renderJSON(el, obj) {
    if (!el) return;
    el.textContent = JSON.stringify(obj, null, 2);
  }

  // ------------------------------------------------------------
  // EVENT FIRE → CBORG ORCHESTRATOR → UI UPDATE
  // ------------------------------------------------------------
  function fireEvent(type) {
    const payload = { ui: true, type };

    const result = CBORG_ORCHESTRATOR.orchestrate(type, payload);

    renderJSON(logEl, result.event);
    renderJSON(dispatchEl, result.dispatch);
    renderJSON(gatewayEl, result.gateway_result);
    renderJSON(stateEl, result.cborg_state);
  }

  // ------------------------------------------------------------
  // BUTTON WIRING
  // ------------------------------------------------------------
  function wireButtons() {
    $$("[data-cyborg-event]").forEach(btn => {
      btn.addEventListener("click", () => {
        const type = btn.getAttribute("data-cyborg-event");
        fireEvent(type);
      });
    });
  }

  // ------------------------------------------------------------
  // INIT
  // ------------------------------------------------------------
  function init() {
    bindElements();
    wireButtons();
  }

  document.addEventListener("DOMContentLoaded", init);

  return {
    fireEvent,
    init
  };
})();
